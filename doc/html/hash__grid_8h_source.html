18  void build(PositionFn positions, size_t num_photons, float radius) {
19  radius_sqr = radius * radius;
20  inv_size = 0.5f / radius;
21 
22  // Compute the global bounding box encompassing all the photons
23  bbox = BBox::empty();
24 
25  #pragma omp parallel for reduction(bbox_extend: bbox)
26  for (size_t i = 0; i < num_photons; ++i)
27  bbox = extend(bbox, positions(i));
28 
29  // Slightly enlarge the bounding box of the photons to avoid numerical problems
30  auto extents = bbox.max - bbox.min;
31  bbox.max += extents * 0.001f;
32  bbox.min -= extents * 0.001f;
33 
34  photons.resize(num_photons);
35  cell_counts.resize(1 << (closest_log2(num_photons) + 1));
36  std::fill(cell_counts.begin(), cell_counts.end(), 0);
37 
38  // Count the number of photons per cell
39 #ifdef USE_STD_THREAD
40  // C++20 version with atomic_ref
41  parallel_for(0, num_photons, [&](size_t i) {
42  auto h = hash_photon(positions(i));
43  std::atomic_ref<uint32_t>(cell_counts[h])++;
44  });
45 #else
46  #pragma omp parallel for
47  for (size_t i = 0; i < num_photons; i++) {
48  auto h = hash_photon(positions(i));
49  #pragma omp atomic
50  cell_counts[h]++;
51  }
52 #endif
53 
54  // Compute the insertion position for each cell
55  std::partial_sum(cell_counts.begin(), cell_counts.end(), cell_counts.begin());
56  assert(cell_counts.back() == photons.size());
57 
58  // Put the photons in their respective cells
59 #ifdef USE_STD_THREAD
60  // C++20 version with atomic_ref
61  parallel_for(0, num_photons, [&](size_t i) {
62  auto h = hash_photon(positions(i));
63  int old_count;
64  old_count = --std::atomic_ref<uint32_t>(cell_counts[h]);
65  photons[old_count] = uint32_t(i);
66  });
67 #else
68  #pragma omp parallel for
69  for (size_t i = 0; i < num_photons; i++) {
70  auto h = hash_photon(positions(i));
71  int old_count;
72  #pragma omp atomic capture
73  old_count = --cell_counts[h];
74  photons[old_count] = uint32_t(i);
75  }
76 #endif

79  template <typename PositionFn, typename InsertFn>
80  void query(const float3& pos, PositionFn positions, InsertFn insert) const {
81  if (!is_inside(bbox, pos)) return;
82 
83  auto p = (pos - bbox.min) * inv_size;
84  int px1 = p.x;
85  int py1 = p.y;
86  int pz1 = p.z;
87  int px2 = px1 + (p.x - px1 > 0.5f ? 1 : -1);
88  int py2 = py1 + (p.y - py1 > 0.5f ? 1 : -1);
89  int pz2 = pz1 + (p.z - pz1 > 0.5f ? 1 : -1);
90 
91  for (int i = 0; i < 8; i++) {
92  auto range = cell_range(i & 1 ? px2 : px1,
93  i & 2 ? py2 : py1,
94  i & 4 ? pz2 : pz1);
95 
96  for (auto j = range.first; j < range.second; j++) {
97  int photon_id = photons[j];
98  auto photon_pos = positions(photon_id);
99  auto d = lensqr(pos - photon_pos);
100  if (d < radius_sqr)
101  insert(photon_id, d);
102  }
103  }
104  }
