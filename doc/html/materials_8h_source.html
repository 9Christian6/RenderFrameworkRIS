51 inline float shading_normal_adjoint(const float3& in, const SurfaceParams& surf, const float3& out) {
52  auto n = std::fabs(dot(in, surf.face_normal) * dot(out, surf.coords.n));
53  auto d = std::fabs(dot(out, surf.face_normal) * dot(in, surf.coords.n));
54  return d != 0 ? n / d : 0.0f;
55 }

180 public:
181  GlassBsdf(float n1 = 1.0f, float n2 = 1.4f, const rgb& ks = rgb(1.0f), const rgb& kt = rgb(1.0f))
182  : Bsdf(Type::Specular)

184  , ks(ks)
185  , kt(kt)
186  {}

188  BsdfSample sample(Sampler& sampler, const SurfaceParams& surf, const float3& out, bool adjoint) const override final {
189  auto k = surf.entering ? eta : 1.0f / eta;
190  auto cos_i = dot(out, surf.coords.n);
191  auto cos2_t = 1.0f - k * k * (1.0f - cos_i * cos_i);
192  if (cos2_t > 0) {
193  // Refraction
194  auto cos_t = std::sqrt(cos2_t);
195  auto F = fresnel_factor(k, cos_i, cos_t);
196  if (sampler() > F) {
197  auto t = (k * cos_i - cos_t) * surf.coords.n - k * out;
198  auto adjoint_term = adjoint ? k * k : 1.0f;
199  return make_sample<true>(t, 1.0f, kt * adjoint_term, surf);
200  }
201  }
202 
203  // Reflection
204  return make_sample(reflect(out, surf.coords.n), 1.0f, ks, surf);
205  }
206 
207 private:
209  static float fresnel_factor(float k, float cos_i, float cos_t) {
210  const float R_s = (k * cos_i - cos_t) / (k * cos_i + cos_t);
211  const float R_p = (cos_i - k * cos_t) / (cos_i + k * cos_t);
212  return (R_s * R_s + R_p * R_p) * 0.5f;
213  }
214 
215  float eta;
216  rgb ks, kt;
217 };

235  sample.color = lerp(use_b ? a->eval(sample.in, surf, out) * cos : sample.color, use_b ? sample.color : b->eval(sample.in, surf, out) * cos, k);
